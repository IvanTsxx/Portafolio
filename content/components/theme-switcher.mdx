```tsx
"use client";

import { useTheme } from "next-themes";
import { useEffect, useState } from "react";

import { MonitorIcon, MoonStarIcon, SunIcon } from "lucide-react";

import { cn } from "@/shared/lib/utils";

export function ThemeSwitcher() {
  const [mounted, setMounted] = useState(false);
  const { theme, setTheme } = useTheme();

  useEffect(() => {
    setMounted(true);
  }, []);

  if (!mounted) {
    return <div className="h-9 w-[120px]" />;
  }

  const themes = [
    { icon: SunIcon, id: "light", label: "Light" },
    { icon: MonitorIcon, id: "system", label: "System" },
    { icon: MoonStarIcon, id: "dark", label: "Dark" },
  ];

  return (
    <div className="inline-flex items-center gap-1 overflow-hidden rounded-full border border-border bg-secondary/50 p-1 backdrop-blur-sm">
      {themes.map((t) => {
        const Icon = t.icon;
        const isActive = theme === t.id;
        return (
          <button
            key={t.id}
            onClick={() => setTheme(t.id)}
            className={cn(
              "flex items-center justify-center rounded-full p-1.5 transition-all duration-200",
              isActive
                ? "bg-background text-foreground shadow-sm"
                : "text-muted-foreground hover:bg-secondary/80 hover:text-foreground"
            )}
            title={t.label}
          >
            <Icon className="h-4 w-4" />
          </button>
        );
      })}
    </div>
  );
}
```
